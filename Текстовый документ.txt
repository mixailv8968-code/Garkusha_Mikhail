        // ========== НАСТРОЙКИ ==========
        const SCROLL_SPEED = 50;
        const SCROLL_DELAY = 300;
        const EXTRA_DISTANCE = 0.1;
        const SCROLL_THRESHOLD = 20;

        function initScrollOnHover() {
            const lines = document.querySelectorAll('.letter-sub-line');
            lines.forEach(line => {
                const textSpan = line.querySelector('.scroll-text');
                if (!textSpan) return;

                let isHovered = false;
                let scrollTimer = null;

                function startScroll() {
                    if (isHovered) return;
                    isHovered = true;

                    textSpan.style.transition = 'none';
                    textSpan.style.transform = '';

                    scrollTimer = setTimeout(() => {
                        if (!isHovered) return;
                        const containerWidth = line.offsetWidth;
                        const textWidth = textSpan.offsetWidth;
                        if (textWidth <= containerWidth) {
                            isHovered = false;
                            return;
                        }
                        let scrollDistance = textWidth - containerWidth;
                        scrollDistance = scrollDistance * (1 + EXTRA_DISTANCE);
                        const duration = scrollDistance / SCROLL_SPEED;
                        textSpan.style.transition = `transform ${duration}s ease`;
                        textSpan.style.transform = `translateX(-${scrollDistance}px)`;
                    }, SCROLL_DELAY);
                }

                function resetScroll() {
                    if (scrollTimer) {
                        clearTimeout(scrollTimer);
                        scrollTimer = null;
                    }
                    if (!isHovered) return;
                    isHovered = false;

                    const currentTransform = textSpan.style.transform;
                    let currentShift = 0;
                    if (currentTransform) {
                        const match = currentTransform.match(/translateX\(-([\d.]+)px\)/);
                        if (match) currentShift = parseFloat(match[1]);
                    }
                    if (currentShift > 0) {
                        const duration = currentShift / SCROLL_SPEED;
                        textSpan.style.transition = `transform ${duration}s ease`;
                        textSpan.style.transform = `translateX(0)`;
                    } else {
                        textSpan.style.transition = '';
                        textSpan.style.transform = '';
                    }
                }

                line.addEventListener('mouseenter', startScroll);
                line.addEventListener('mouseleave', resetScroll);

                window.addEventListener('resize', () => {
                    if (!isHovered) {
                        textSpan.style.transition = '';
                        textSpan.style.transform = '';
                    } else {
                        textSpan.style.transition = '';
                        textSpan.style.transform = '';
                        isHovered = false;
                        if (scrollTimer) clearTimeout(scrollTimer);
                        startScroll();
                    }
                });
            });
        }

        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', initScrollOnHover);
        } else {
            initScrollOnHover();
        }

        // ========== КОД БЕСКОНЕЧНОЙ КАРУСЕЛИ ==========
        (function() {
            const track = document.getElementById('carouselTrack');
            const originalSlides = Array.from(document.querySelectorAll('.carousel-slide'));
            const prevBtn = document.getElementById('prevBtn');
            const nextBtn = document.getElementById('nextBtn');
            const overlay = document.getElementById('imageOverlay');
            const overlayImg = overlay.querySelector('img');

            if (!originalSlides.length) return;

            const firstClone = originalSlides[0].cloneNode(true);
            const lastClone = originalSlides[originalSlides.length - 1].cloneNode(true);
            track.insertBefore(lastClone, originalSlides[0]);
            track.appendChild(firstClone);

            const allSlides = document.querySelectorAll('.carousel-slide');
            const totalSlides = allSlides.length;
            let currentIndex = 1;
            let slideWidth, gap, containerWidth, step;
            let isDragging = false;
            let startX = 0;
            let currentTranslate = 0;
            let startTranslate = 0;
            let dragStartTime = 0;
            let dragHappened = false;
            const DRAG_THRESHOLD = 5;

            function updateMeasurements() {
                slideWidth = allSlides[0].offsetWidth;
                gap = parseFloat(getComputedStyle(track).gap) || 0;
                containerWidth = document.querySelector('.carousel-container').offsetWidth;
                step = slideWidth + gap;
            }

            function getOffsetForIndex(index) {
                const slideLeft = index * step;
                return (containerWidth / 2) - (slideLeft + slideWidth / 2);
            }

            function updatePosition(animate = true) {
                if (!animate) {
                    track.style.transition = 'none';
                } else {
                    track.style.transition = 'transform 0.4s ease';
                }
                const offset = getOffsetForIndex(currentIndex);
                track.style.transform = `translateX(${offset}px)`;
                currentTranslate = offset;
                allSlides.forEach((slide, i) => {
                    slide.classList.toggle('active', i === currentIndex);
                });
                if (!animate) {
                    track.offsetHeight;
                    track.style.transition = 'transform 0.4s ease';
                }
            }

            function goToSlide(newIndex, animated = true) {
                if (newIndex < 0 || newIndex >= totalSlides) return;
                currentIndex = newIndex;
                updatePosition(animated);
                handleInfinite();
            }

            function handleInfinite() {
                if (currentIndex === 0) {
                    setTimeout(() => {
                        currentIndex = totalSlides - 2;
                        updatePosition(false);
                    }, 400);
                } else if (currentIndex === totalSlides - 1) {
                    setTimeout(() => {
                        currentIndex = 1;
                        updatePosition(false);
                    }, 400);
                }
            }

            function onDragStart(e) {
                e.preventDefault();
                startX = e.type === 'mousedown' ? e.pageX : e.touches[0].pageX;
                isDragging = true;
                dragHappened = false;
                track.classList.add('dragging');
                dragStartTime = Date.now();
                startTranslate = currentTranslate;
                track.style.transition = 'none';
            }

            function onDragMove(e) {
                if (!isDragging) return;
                e.preventDefault();
                const currentX = e.type === 'mousemove' ? e.pageX : e.touches[0].pageX;
                const deltaX = currentX - startX;
                const newTranslate = startTranslate + deltaX;
                track.style.transform = `translateX(${newTranslate}px)`;
                currentTranslate = newTranslate;

                if (Math.abs(deltaX) > DRAG_THRESHOLD) {
                    dragHappened = true;
                }
            }

            function onDragEnd(e) {
                if (!isDragging) return;
                isDragging = false;
                track.classList.remove('dragging');
                const dragTime = Date.now() - dragStartTime;
                const distance = currentTranslate - startTranslate;
                const speed = Math.abs(distance) / dragTime;
                const threshold = slideWidth * 0.3;
                let deltaIndex = 0;
                if (Math.abs(distance) > threshold) {
                    deltaIndex = distance > 0 ? -1 : 1;
                } else if (speed > 0.5) {
                    deltaIndex = distance > 0 ? -1 : 1;
                }
                let newIndex = currentIndex + deltaIndex;
                if (newIndex < 0) newIndex = 0;
                if (newIndex >= totalSlides) newIndex = totalSlides - 1;
                goToSlide(newIndex, true);

                setTimeout(() => {
                    dragHappened = false;
                }, 200);
            }

            track.addEventListener('mousedown', onDragStart);
            window.addEventListener('mousemove', onDragMove);
            window.addEventListener('mouseup', onDragEnd);

            track.addEventListener('touchstart', onDragStart);
            window.addEventListener('touchmove', onDragMove, { passive: false });
            window.addEventListener('touchend', onDragEnd);

            prevBtn.addEventListener('click', () => goToSlide(currentIndex - 1));
            nextBtn.addEventListener('click', () => goToSlide(currentIndex + 1));

            window.addEventListener('resize', () => {
                updateMeasurements();
                updatePosition(false);
            });

            updateMeasurements();
            updatePosition(false);

            track.addEventListener('click', (e) => {
                if (dragHappened) return;
                const slide = e.target.closest('.carousel-slide');
                if (!slide) return;
                const img = slide.querySelector('img');
                if (!img) return;
                overlayImg.src = img.src;
                overlay.style.display = 'flex';
            });

            overlay.addEventListener('click', (e) => {
                if (e.target === overlay || e.target === overlayImg) {
                    overlay.style.display = 'none';
                }
            });
        })();

        (function() {
            const strip = document.querySelector('.vertical-strip-secondary');
            const mLink = document.querySelector('.letter-m');
            const subContainer = document.querySelector('.letter-sub');

            if (!strip || !mLink || !subContainer) return;

            function addEffects() {
                strip.classList.add('hover');
                subContainer.classList.add('expanded');
                mLink.classList.add('extra-letters');
            }

            function removeEffects() {
                strip.classList.remove('hover');
                subContainer.classList.remove('expanded');
                mLink.classList.remove('extra-letters');
            }

            strip.addEventListener('mouseenter', addEffects);
            strip.addEventListener('mouseleave', removeEffects);
            mLink.addEventListener('mouseenter', addEffects);
            mLink.addEventListener('mouseleave', removeEffects);
            subContainer.addEventListener('mouseenter', addEffects);
            subContainer.addEventListener('mouseleave', removeEffects);
        })();

        const nextBtn = document.getElementById('nextProjectBtn');
        let targetUrl = '#';

        function updateTargetUrl() {
            const links = document.querySelectorAll('.letter-sub-line');
            if (links.length === 0) return;
            let boldIndex = -1;
            for (let i = 0; i < links.length; i++) {
                if (links[i].classList.contains('bold')) {
                    boldIndex = i;
                    break;
                }
            }
            if (boldIndex !== -1 && boldIndex + 1 < links.length) {
                const nextLink = links[boldIndex + 1];
                if (nextLink.href && nextLink.href !== '#') {
                    targetUrl = nextLink.href;
                }
            }
        }
        updateTargetUrl();

        window.addEventListener('scroll', function() {
            if (window.innerHeight + window.scrollY >= document.documentElement.scrollHeight - SCROLL_THRESHOLD) {
                nextBtn.classList.add('visible');
            } else {
                nextBtn.classList.remove('visible');
            }
        });

        nextBtn.addEventListener('click', function() {
            if (targetUrl && targetUrl !== '#') {
                window.location.href = targetUrl;
            }
        });
